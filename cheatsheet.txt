list
----
docker ps
docker ps -a
docker images
docker volume ls
docker network ls

stop
----
docker stop <container>

remove
------
docker rm <container>
docker rmi <image>
docker volume prune   (Removes all unused volumes)
docker network rm <network_name>
docker system prune  (Remove unused containers, images, networks, build cache, does not remove volumes and used images)
docker system prune -af --volumes (Remove everything)


build image
-----------
docker build -t <image_name> .  
(-t means tag <name>:<tag> like nginx:v1, 
if only nginx, docker will auto assign latest tag)

inspect 
--------
docker volume inspect <volume_name>
docker network inspect <network_name>

docker compose
---------------
docker compose up 
docker compose up --build (rebuilds images)
docker compose up --force-recreate (recreates containers even if nothing changed)
docker compose build
docker compose down (removes containers and networks, optional for images and volumes)
docker compose down -v
docker compose down --rmi all --volumes --remove-orphans
docker compose logs
docker compose logs -f
docker compose exec wordpress bash
docker compose exec mariadb sh
docker compose restart nginx
docker compose ps

==========================
DOCKER & DOCKER COMPOSE FLAGS CHEAT SHEET
==========================

--- DOCKER FLAGS ---

-v / --volumes
  - docker compose down -v  → remove volumes created by Compose project
  - docker run -v <host>:<container> → bind mount host directory or create volume

-f / --force
  - docker system prune -f → force removal without confirmation prompt

-a / --all
  - docker system prune -a → remove all unused images, not just dangling
  - docker images -a → list all images, including intermediate/dangling

-p / --publish
  - docker run -p <host_port>:<container_port> → map host port to container port

-d / --detach
  - docker run -d → run container in background

--name <name>
  - Assign a name to a container

--rm
  - Remove container automatically after exit

-it
  - Interactive + TTY (attach terminal)

--network <network>
  - Connect container to a specific network

--build
  - Build image before starting containers (Docker Compose up --build)

---

--- DOCKER COMPOSE FLAGS ---

-f <file>
  - Specify compose file instead of default docker-compose.yml

--build
  - Build images before starting containers (docker compose up --build)

-v / --volumes
  - Remove volumes created by the Compose project (docker compose down -v)

--rmi all
  - Remove images built by the Compose project (docker compose down --rmi all)

--remove-orphans
  - Remove containers not defined in the current compose file

-d
  - Detached mode (run containers in background)

logs -f
  - Follow logs in real-time

---

--- PRUNE FLAGS (Docker Cleanup) ---

docker system prune
  - Remove unused containers, networks, dangling images, build cache
  - Does NOT remove volumes or images still used by containers

docker system prune -a
  - Remove all unused images (not just dangling)

docker system prune -af --volumes
  - Remove EVERYTHING unused including:
    - stopped containers
    - all unused images
    - all build cache
    - all networks
    - all volumes (MariaDB, WordPress, etc.)

docker volume prune
  - Remove all unused volumes

docker network prune
  - Remove all unused networks
